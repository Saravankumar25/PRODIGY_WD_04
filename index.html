4. Check if a Number is Prime


int n = 17;
boolean isPrime = n > 1;
for (int i = 2; i <= Math.sqrt(n); i++) {
    if (n % i == 0) {
        isPrime = false;
        break;
    }
}
System.out.println(n + " is " + (isPrime ? "Prime" : "Not Prime"));

5. Print All Prime Numbers in a Range

    boolean prime = i > 1;
    for (int j = 2; j <= Math.sqrt(i); j++) {
        if (i % j == 0) {
            prime = false;
            break;
        }
    }
    if (prime) System.out.print(i + " ");
}
System.out.println();

29. Pascal's Triangle


int rows = 5;
for (int i = 0; i < rows; i++) {
    int val = 1;
    for (int space = 1; space < rows - i; space++)
        System.out.print(" ");
    for (int j = 0; j <= i; j++) {
        System.out.print(val + " ");
        val = val * (i - j) / (j + 1);
    }
    System.out.println();
}

39. Binary Search (Array Must Be Sorted)


int[] arr = {1, 3, 5, 6, 9};
int target = 6;
int low = 0, high = arr.length - 1;
boolean isFound = false;
while (low <= high) {
    int mid = (low + high) / 2;
    if (arr[mid] == target) {
        isFound = true;
        System.out.println("Found at index " + mid);
        break;
    } else if (arr[mid] < target) {
        low = mid + 1;
    } else {
        high = mid - 1;
    }
}
if (!isFound)
    System.out.println("Not found");
